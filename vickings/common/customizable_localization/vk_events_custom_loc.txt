GetLegVoteResultsBill = {
	type = character
	text = {
		trigger = {
			primary_title = {
				exists = var:yay_predict
				exists = var:nay_predict
				var:yay_predict > var:nay_predict
			}
		}
		localization_key = propbill.PASS
	}
	text = {
		trigger = {
			primary_title = {
				exists = var:yay_predict
				exists = var:nay_predict
				var:yay_predict = var:nay_predict
			}
		}
		localization_key = propbill.TIE
	}
	text = {
		localization_key = propbill.FAIL
		fallback = yes
	}
}
