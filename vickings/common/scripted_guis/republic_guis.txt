can_pass_four_year_limit = {
	scope = character
	is_valid = {
		holds_valid_merchant_republic = yes
		custom_description = {
			text = "recently_passed_reform"
			subject = root
			primary_title = {
				NOT = { has_variable = recent_reform_passed }
			}
		}
		custom_description = {
			text = "law_under_debate"
			subject = root
			primary_title = {
				NOT = { has_variable = debating_law }
			}
		}
		primary_title = {
			NOT = { is_target_in_variable_list = { name = enacted_laws_list target = flag:four_year_term } }
		}
	}
	effect = {
		custom_tooltip = enact_term_limit
		if = {
			limit = { is_independent_ruler = yes }
			primary_title = {
				set_variable = { name = debating_law value = flag:four_year_term }
				generate_party_acceptance_values = {
					SOCIAL_LIBERTY = 10
					SOCIAL_HIERARCHY = 10 
					LAISSEZ_FAIRE = 0 REG_ECONOMY = 0 PRO_SLAVERY = 0 UNIVERSAL_ACCEPTANCE_VALUE = 0
				}
			}
			trigger_event = legislature.1
		}
		else = {
			primary_title = {
				clear_term_limit_reforms = yes
				add_to_variable_list = { name = enacted_laws_list target = flag:four_year_term }
				set_variable = { name = recent_reform_passed value = flag:recent_reform_passed years = 5 }
			}
		}
	}
}

					}
